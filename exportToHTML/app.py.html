from flask import Flask, render_template, request, jsonify

app = Flask(__name__)

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/compare', methods=['POST'])
def compare():
    # This is where you handle the comparison logic
    # For now, just returning a simple JSON response
    product_addresses = request.form.getlist('product_addresses')
    user_requirements = request.form.get('user_requirements')
    return jsonify({
        'product_addresses': product_addresses,
        'user_requirements': user_requirements,
        'result': 'Comparison results will be here'
    })

if __name__ == '__main__':
    app.run(debug=True)
